# Document not found (404)

This URL is invalid, sorry. Please use the navigation bar or search to continue. (test)

<script>
  (function() {
    const path = window.location.pathname;
    const pathParts = path.split('/').filter(p => p.length > 0);

    // This assumes a URL structure of /<repo-name>/<lang-code>/...
    if (pathParts.length > 2) {
      const langCode = pathParts[1];
      if (langCode.length >= 2 && langCode.length <= 5) {
        // Construct the path to the translated 404 page, e.g., /comprehensive-rust/es/404.html
        const basePath = `/${pathParts[0]}/`;
        const translated404Url = `${basePath}${langCode}/404.html`;

        fetch(translated404Url)
          .then(response => {
            if (response.ok) {
              return response.text();
            }
            throw new Error(`Translated 404 page not found at ${translated404Url}`);
          })
          .then(html => {
            const parser = new DOMParser();
            const translatedDoc = parser.parseFromString(html, 'text/html');
            const newContent = translatedDoc.querySelector('main');
            const newTitle = translatedDoc.querySelector('title');

            if (newContent) {
              document.querySelector('main').innerHTML = newContent.innerHTML;
            }
            if (newTitle) {
              document.title = newTitle.innerText;
            }
          })
          .catch(error => {
            console.warn("Failed to load translated 404 page, showing default.", error);
          });
      }
    }
  })();
</script>
